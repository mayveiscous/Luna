cmake_minimum_required(VERSION 3.15)
project(luna LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE RelWithDebInfo)
endif()

# Lua 5.3 via pkg-config. CMake's bundled FindLua module doesn't reliably
# pick the right version when multiple Lua dev packages are installed
# side by side (as Debian/Ubuntu allow); pkg-config's lua5.3.pc is exact.
find_package(PkgConfig REQUIRED)
pkg_check_modules(LUA REQUIRED lua5.4)

find_package(Python3 REQUIRED COMPONENTS Development)

add_library(luna_python MODULE
    src/module.cpp
    src/python_runtime.cpp
    src/py_object.cpp
    src/conversions.cpp
    src/errors.cpp
    src/bridge.cpp
)

target_include_directories(luna_python PRIVATE
    include
    ${LUA_INCLUDE_DIRS}
    ${Python3_INCLUDE_DIRS}
)

target_link_libraries(luna_python PRIVATE
    ${LUA_LIBRARIES}
    Python3::Python
    ${CMAKE_DL_LIBS}
)

target_compile_options(luna_python PRIVATE ${LUA_CFLAGS_OTHER} -Wall -Wextra)

# Embedding CPython inside a module that is itself dlopen'd (by Lua's
# require()) means libpython's symbols aren't promoted to the global
# symbol scope the way they would be for a normal `python3` executable.
# Python C-extensions (numpy, etc.) are internally dlopen'd too, and need
# to resolve CPython C API symbols against something already loaded
# globally -- otherwise imports fail with "undefined symbol:
# PyObject_SelfIter" or similar. runtime_init() works around this by
# re-opening libpython itself with RTLD_GLOBAL before Py_Initialize();
# this define tells it exactly which file to open.
target_compile_definitions(luna_python PRIVATE
    LUNA_LIBPYTHON_PATH="${Python3_LIBRARIES}"
)

# Lua's require() looks for a symbol named luaopen_<modname> in a file
# named <modname>.so (no "lib" prefix), so the output must be exactly
# "python.so", not "libluna_python.so".
set_target_properties(luna_python PROPERTIES
    OUTPUT_NAME "python"
    PREFIX ""
)

if(APPLE)
    set_target_properties(luna_python PROPERTIES SUFFIX ".so")
endif()
