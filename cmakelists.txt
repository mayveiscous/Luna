cmake_minimum_required(VERSION 3.15)
project(luna LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE RelWithDebInfo)
endif()

find_package(PkgConfig REQUIRED)
pkg_check_modules(LUA REQUIRED lua5.4)

find_package(Python3 REQUIRED COMPONENTS Development)

add_library(luna_python MODULE
    src/module.cpp
    src/python_runtime.cpp
    src/py_object.cpp
    src/conversions.cpp
    src/errors.cpp
    src/bridge.cpp
)

target_include_directories(luna_python PRIVATE
    include
    ${LUA_INCLUDE_DIRS}
    ${Python3_INCLUDE_DIRS}
)

target_link_libraries(luna_python PRIVATE
    ${LUA_LIBRARIES}
    Python3::Python
    ${CMAKE_DL_LIBS}
)

target_compile_options(luna_python PRIVATE ${LUA_CFLAGS_OTHER} -Wall -Wextra)

target_compile_definitions(luna_python PRIVATE
    LUNA_LIBPYTHON_PATH="${Python3_LIBRARIES}"
)

set_target_properties(luna_python PROPERTIES
    OUTPUT_NAME "python"
    PREFIX ""
)

if(APPLE)
    set_target_properties(luna_python PROPERTIES SUFFIX ".so")
endif()
